[source,bash]
.docker-compose.yml
----
version: '3'
services:
  db:
    image: postgres
  web:
    build: .
    command: bundle exec rails s -p 3000 -b '0.0.0.0'
    volumes:
      - .:/app
    ports:
      - "3000:3000"
    depends_on:
      - db
----

docker ps -a
docker commit -a "@wbzyl" -m "Set CMD git" command-git ubuntu-git

docker run --name command-git ubuntu-git version
